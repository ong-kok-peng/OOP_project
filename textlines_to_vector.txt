#include <iostream>
#include <vector>
#include <string>
using namespace std;

vector<vector<string>> freightinfo;

// Base class
class Freight {
  public: 
    Freight(string id, string country, string time) {
    	freightinfo.push_back({id, country, time});
    }
    
    void showFreight() {
    	for (int a=0; a < freightinfo.size(); a++) {
        	for (int b = 0; b < freightinfo[a].size(); b++) {
                cout << freightinfo[a][b] << ", ";
            }
            cout << "\n";
        }
    }
};

int main() {
  Freight theFreight("F01", "Singapore", "0506");
  Freight theFreight1("F02", "Japan", "0623");
  Freight theFreight2("F03", "United States", "0745");
  Freight theFreight3("F05", "North Korea", "2359");
  
  theFreight.showFreight();
  return 0;
}
